상속과 다형성

class Person {
	var id = 0
	var name = "이름"
	var email = "abc@gmail.com"

	func walk() {
		print("사람이 걷는다.")
	}
}

class Student: Person {
	var studentId = 1

	override walk() { // 재정의 메서드
		print("학생이 걷는다.")
	}

	func study() {
		print("학생이 공부한다.")
	}
}

let person = Person()
person.walk() // "사람이 걷는다."

let student: Person = Student() // 업캐스팅
student.walk() // "학생이 걷는다."
// 업캐스팅한 Person의 walk() 메서드가 실행되지 않고
// 원래의 Student의 walk() 메서드가 실행되었다.

다형성(Polymorphism)에 대한 이해

  • 스택 영역에 인스턴스를 가리키는 변수나 상수가 있다.
  • 인스턴스는 힙 공간에 캐스팅 이전의 원래의 형태로 저장되어 있다.
  • 힙 영역의 인스턴스는 데이터 영역에 있는 본인의 클래스를 가리킨다.

다형성 (Polymorphism)

  • 하나의 인스턴스가 여러가지 형태로 표현될 수 있다.
    • 하나의 타입을 여러가지 형태로 해석할 수 있다.
  • 다향성의 구현은 클래스의 상속과 깊게 연관되어 있다.
    • 이후 배울 protocol 또한 연관되어 있음
